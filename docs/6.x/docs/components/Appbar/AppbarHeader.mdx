A component to use as a header at the top of the screen.
It can contain the screen title, controls such as navigation buttons, menu button etc.




  <ScreenshotTabs screenshotData={{"small":"screenshots/appbar-small.png","medium":"screenshots/appbar-medium.png","large":"screenshots/appbar-large.png","center-aligned":"screenshots/appbar-center-aligned.png"}} baseUrl="/react-native-paper/"/>


## Usage
```js
import * as React from 'react';
import { Appbar } from 'react-native-paper';

const MyComponent = () => {
  const _goBack = () => console.log('Went back');

  const _handleSearch = () => console.log('Searching');

  const _handleMore = () => console.log('Shown more');

  return (
    <Appbar.Header>
      <Appbar.BackAction onPress={_goBack} />
      <Appbar.Content title="Title" />
      <Appbar.Action icon="magnify" onPress={_handleSearch} />
      <Appbar.Action icon="dots-vertical" onPress={_handleMore} />
    </Appbar.Header>
  );
};

export default MyComponent;
```
