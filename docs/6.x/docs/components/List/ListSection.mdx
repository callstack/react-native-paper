A component used to group list items.




  <ScreenshotTabs screenshotData={"screenshots/list-section.png"} baseUrl="/react-native-paper/"/>


## Usage
```js
import * as React from 'react';
import { List, Palette } from 'react-native-paper';

const MyComponent = () => (
  <List.Section>
    <List.Subheader>Some title</List.Subheader>
    <List.Item title="First Item" left={() => <List.Icon icon="folder" />} />
    <List.Item
      title="Second Item"
      left={() => <List.Icon color={Palette.tertiary70} icon="folder" />}
    />
  </List.Section>
);

export default MyComponent;
```
