## Usage
```js
import * as React from 'react';
import { Icon, MD3Colors } from 'react-native-paper';

const MyComponent = () => (
  <Icon
    source="camera"
    color={MD3Colors.error50}
    size={20}
  />
);

export default MyComponent;
```
